48 void Parser::tokenizeConfig(string line) {
49  while (line.size() > 0) {
50  size_t pos = line.find_first_not_of(" \t");
51  if (pos == line.npos || line[pos] == '#') {
52  break;
53  }
54  line = line.substr(pos);
55  pos = line.find_first_of("\\#{}; \t\n\0");
56  pos == 0 ? pos = 1 : pos;
57  string tmp = line.substr(0, pos);
58  tokens.push_back(tmp);
59  ;
60  line = line.substr(tmp.size());
61  }
62 }
